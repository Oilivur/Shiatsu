<template>
  <header>
    <router-link to="/" class="logo">Terindo</router-link>

    <!-- Burger menu -->
    <div class="icons" @click="toggleMenu">
      <i class="bx bx-menu" v-if="!menuOpen"></i>
      <i class="bx bx-x" v-else></i>
    </div>

    <!-- Animated dropdown menu with staggered effect -->
    <transition name="menu-fade">
      <nav v-if="menuOpen" class="mobile-nav">
        <transition-group name="dropdown" tag="div">
          <router-link v-for="(item, index) in menuItems" 
            :key="index" 
            :to="item.link"
            :style="{ transitionDelay: `${index * 100}ms` }"
          >
            {{ item.label }}
          </router-link>
        </transition-group>
      </nav>
    </transition>
  </header>
</template>

<script>
import { inject, ref, onMounted, onUnmounted } from "vue";

export default {
  setup() {
    const isAdmin = ref(false);
    const handleLogout = inject("handleLogout");
    const menuOpen = ref(false);

    const toggleMenu = () => {
      menuOpen.value = !menuOpen.value;
    };

    const checkScreenSize = () => {
      if (window.innerWidth > 1079) {
        menuOpen.value = true;
      } else {
        menuOpen.value = false;
      }
    };

    onMounted(() => {
      window.addEventListener("resize", checkScreenSize);
      const storedKey = localStorage.getItem("adminKey");
      if (storedKey) {
        isAdmin.value = true;
      }
    });

    onUnmounted(() => {
      window.removeEventListener("resize", checkScreenSize);
    });

    const menuItems = [
      { label: "Teraapiad-massaažid", link: "/termas" },
      { label: "Shindo treeningud", link: "/shindo" },
      { label: "Hinnakiri", link: "/hinnakiri" },
      { label: "Minust", link: "/minust" },
      { label: "Kontakt", link: "/kontakt" }
    ];

    return { isAdmin, handleLogout, menuOpen, toggleMenu, menuItems };
  }
};
</script>

<style scoped>
/* Burger Menu */
.icons {
  display: none;
  cursor: pointer;
  font-size: 2rem;
  color: blanchedalmond;
}

@media (max-width: 1079px) {
  .icons {
    display: inline-flex;
  }

  .mobile-nav {
    display: flex;
    flex-direction: column;
    align-items: center;
    gap: 0.5rem;
    height: auto;
    position: absolute;
    top: 100%;
    left: 0;
    width: 100%;
    background: rgba(0, 0, 0, 0.1);
    backdrop-filter: blur(50px);
    box-shadow: 0 .5rem 1rem rgba(0, 0, 0, .1);
    text-align: center;
    padding: 1rem 0;
  }

  /* Each menu item appears one by one */
  .dropdown-enter-active {
    transition: opacity 0.3s ease-out, transform 0.3s ease-out;
  }

  .dropdown-enter-from {
    opacity: 0;
    transform: translateY(-10px);
  }

  .dropdown-enter-to {
    opacity: 1;
    transform: translateY(0);
  }

  .dropdown-leave-active {
    transition: opacity 0.2s ease-in, transform 0.2s ease-in;
  }

  .dropdown-leave-from {
    opacity: 1;
    transform: translateY(0);
  }

  .dropdown-leave-to {
    opacity: 0;
    transform: translateY(-10px);
  }

  /* Full menu fades in and out */
  .menu-fade-enter-active, .menu-fade-leave-active {
    transition: opacity 0.4s ease-in-out;
  }

  .menu-fade-enter-from, .menu-fade-leave-to {
    opacity: 0;
  }
}
</style>
